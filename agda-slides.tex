\begin{frame}[fragile]
  \frametitle{Agda timeline in a nutshell}
  \begin{description}
    \item[1969]: Cornelis Vreeswijk releases the song \emph{H\"onan Agda}%
      \footnote{\url{https://lists.chalmers.se/pipermail/agda/2016/008867.html}}
    \item[1970s]: Martin-L\"of intuitionistic type theory in mathematics 
    \item[1999]: Catarina Coquand implemented in Haskell the original version of Agda
    \item[2007]: Ulf Norell developed Agda 2, with implementation described in his PhD thesis \emph{Towards a practical programming language based on dependent type theory}%
      \footnote{\url{https://www.cse.chalmers.se/~ulfn/papers/thesis.pdf}}
  \end{description}
\end{frame}
